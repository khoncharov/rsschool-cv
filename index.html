Code Example

Count characters in a string. If you have a string like aba, then the result should be {'a': 2, 'b': 1}. 


function count(string) {
  return string.split('').reduce((res, item) => {
    res[item] ? res[item]++ : res[item] = 1
    return res
  }, {})
}
